steal some keywords of haskell perlude to ease manipulations of iterators and stream in perl.

    use Perlude;
    now {say if /false/} take 2, lines "/etc/passwd";

is the same as

    open my $fh,"/etc/passwd";
    my $count = 0;

    while (<$fh>) {
        if (/root/) {
            $count++;
            say;
        }
        last if $count == 2;
    }
